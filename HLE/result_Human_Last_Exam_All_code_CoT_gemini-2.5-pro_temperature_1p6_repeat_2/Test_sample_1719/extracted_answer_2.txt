def print_obstruction_groups():
    """
    Prints the list of homology and homotopy groups that form the obstructions
    for the two paths of bundle automorphisms to be homotopic.
    """
    print("The homotopy-theoretic obstructions are determined by the following groups:")
    print("1. The fundamental group of the special orthogonal group:")
    print("   \pi_1(SO(2k))")
    print("\n2. A homology group of X with coefficients in a homotopy group of SO(2k):")
    print("   H_{n-1}(X; \pi_{1-n}(SO(2k)))")
    print("\n3. Another homology group of X that contributes to the differentials in the calculation:")
    print("   H_{n-1}(X; \pi_{2-n}(SO(2k)))")

print_obstruction_groups()
