void main() {
    // Variable Declarations (Total Memory: 174D)
    // result:     160 * 1D = 160D
    // num_digits: 1   * 3D =   3D
    // i, j, temp: 3   * 3D =   9D
    // carry:      1   * 2D =   2D
    digit result[160];
    char num_digits, i, j, temp;
    short carry;

    // Initialize the result array to all zeros.
    for (j = 0; j < 160; j = j + 1) {
        result[j] = 0;
    }

    // Set the initial value of the number to 1.
    result[0] = 1;
    num_digits = 1;

    // Loop from 2 to 100 to calculate 100!
    for (i = 2; i <= 100; i = i + 1) {
        carry = 0;
        // Multiply the large number in 'result' by 'i'.
        for (j = 0; j < num_digits; j = j + 1) {
            temp = result[j] * i + carry;
            result[j] = temp % 10;
            carry = temp / 10;
        }

        // If there's a remaining carry, extend the number of digits.
        while (carry > 0) {
            result[num_digits] = carry % 10;
            carry = carry / 10;
            num_digits = num_digits + 1;
        }
    }

    // Print the final result.
    printf("100! = ");
    for (j = num_digits - 1; j >= 0; j = j - 1) {
        printf("%d", result[j]);
    }
    printf("\n");
}
