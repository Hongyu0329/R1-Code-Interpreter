import sys

def solve():
    """
    This function calculates the expected time until only one particle remains.
    The user should provide the integer values for N1, M1, N2, M2 as command-line arguments.
    Example: python your_script.py 1 1 1 1
    """
    if len(sys.argv) != 5:
        print("Usage: python your_script.py N1 M1 N2 M2")
        print("Please provide four positive integer arguments.")
        return

    try:
        N1 = int(sys.argv[1])
        M1 = int(sys.argv[2])
        N2 = int(sys.argv[3])
        M2 = int(sys.argv[4])
        if not all(x > 0 for x in [N1, M1, N2, M2]):
            print("Arguments must be positive integers.")
            return
    except ValueError:
        print("All arguments must be integers.")
        return

    # The formula for the total expected time tau is derived from summing the
    # expected times of the two phases of the process (5->3 particles and 3->1 particles).
    # The rates lambda1=1 and lambda2=2 are incorporated into the formula's constants.
    # E[tau] = E[tau1] + E[tau2']
    # E[tau1] = ( (N1+M2)*(M1+N2) ) / 2
    # E[tau2'] = ( (N1+M1)*(N2+M2) ) / 4 + ( N1*M1 + N2*M2 ) / 4

    # Calculation for the first phase
    E_tau1 = (N1 + M2) * (M1 + N2) / 2.0
    
    # Calculation for the second phase
    term1_tau2 = (N1 + M1) * (N2 + M2) / 4.0
    term2_tau2 = (N1 * M1 + N2 * M2) / 4.0
    E_tau2_prime = term1_tau2 + term2_tau2

    # Total expected time
    total_E_tau = E_tau1 + E_tau2_prime

    # Output the result step-by-step
    print(f"The total expected time, E[tau], is calculated as the sum of the expected times of two stages.")
    print(f"1. Expected time for the first collision (5 -> 3 particles), E[tau1]:")
    print(f"   E[tau1] = (({N1} + {M2}) * ({M1} + {N2})) / 2 = {E_tau1}")
    
    print(f"2. Expected time for the second collision (3 -> 1 particle), E[tau2']:")
    print(f"   E[tau2'] = (({N1} + {M1}) * ({N2} + {M2})) / 4 + ({N1} * {M1} + {N2} * {M2}) / 4")
    print(f"   E[tau2'] = {term1_tau2} + {term2_tau2} = {E_tau2_prime}")

    print(f"3. Total expected time, E[tau] = E[tau1] + E[tau2']:")
    print(f"   E[tau] = {E_tau1} + {E_tau2_prime} = {total_E_tau}")


if __name__ == "__main__":
    solve()
