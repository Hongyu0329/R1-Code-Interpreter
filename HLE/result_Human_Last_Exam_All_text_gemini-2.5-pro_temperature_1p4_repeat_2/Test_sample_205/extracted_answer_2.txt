void main() {
    // Variable Declarations - Total Memory: 169D
    digit result[158]; // 158 * 1D = 158D. Stores the digits of 100!.
    char i, j;         // 3D + 3D = 6D. Loop counters.
    short carry;       // 2D. For multiplication carry-over (max value 99).
    char product;      // 3D. Stores intermediate product (max value 999).

    // Initialize result array to all zeros.
    for (j = 0; j < 158; j++) {
        result[j] = 0;
    }

    // Set initial value of result to 1.
    result[0] = 1;

    // Main loop to calculate 100! by multiplying from 2 up to 100.
    for (i = 2; i <= 100; i++) {
        carry = 0;
        // Multiply the current number stored in 'result' by 'i'.
        for (j = 0; j < 158; j++) {
            product = result[j] * i + carry;
            result[j] = product % 10;
            carry = product / 10;
        }
    }

    // Print the final result.
    // First, find the starting (most significant) digit to avoid printing leading zeros.
    j = 157;
    while (j > 0 && result[j] == 0) {
        j--;
    }

    // Print digits from most significant to least significant.
    for (; j >= 0; j--) {
        printf("%d", result[j]);
    }
}
