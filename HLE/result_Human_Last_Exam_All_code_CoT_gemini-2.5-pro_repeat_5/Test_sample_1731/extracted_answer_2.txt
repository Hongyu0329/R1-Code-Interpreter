import sympy as sp

def solve_photon_gas_equilibrium():
    """
    Calculates and displays the equilibrium mean energy and entropy for a photon gas
    based on the principles of statistical mechanics.

    The derivation uses the black-body radiation model for light quanta (photons),
    which are bosons with zero chemical potential. The results are applications of
    the principles formalized by the Boltzmann-Sanov and Cramer-Chernoff theorems.
    """

    # --- Step 1: Define the symbolic variables ---
    # k_B: Boltzmann constant
    # h: Planck's constant
    # c: Speed of light
    # V: Volume of the container
    # T: Absolute temperature
    # pi: Mathematical constant pi
    k_B, h, c, V, T = sp.symbols('k_B h c V T', positive=True, real=True)
    pi = sp.pi

    print("This script calculates the equilibrium values for Mean Energy (U) and Entropy (S) of a photon gas.\n")
    print("The derivation relies on the Bose-Einstein statistics for photons in a volume V at temperature T.\n")

    # --- Step 2: Define the Mean Energy (U) ---
    # The mean energy U of a photon gas is given by the Stefan-Boltzmann law.
    # U = (8 * pi^5 * k_B^4 / (15 * h^3 * c^3)) * V * T^4
    # This result is obtained by integrating the energy of photons over the
    # Bose-Einstein distribution multiplied by the density of states.
    # Integral[0, inf] (epsilon^3 / (exp(epsilon/(k_B*T)) - 1)) d_epsilon = (pi^4 / 15) * (k_B*T)^4
    
    stefan_boltzmann_constant_group = (8 * pi**5 * k_B**4) / (15 * h**3 * c**3)
    U = stefan_boltzmann_constant_group * V * T**4

    print("--- Equilibrium Mean Energy (U) ---")
    print("The mean energy U is derived from integrating the energy over all states:")
    print("U = Integral(epsilon * <n_epsilon> * g(epsilon) d_epsilon)")
    print("This yields the Stefan-Boltzmann energy law for a black body.\n")
    print("The final equation for Mean Energy (U) is:")
    sp.pprint(U, use_unicode=True)
    print("\n" + "="*70 + "\n")

    # --- Step 3: Define the Entropy (S) ---
    # The entropy S can be derived from the thermodynamic relation S = (U - F)/T,
    # where F is the Helmholtz Free Energy. For a photon gas, F = -U/3.
    # Therefore, S = (U - (-U/3))/T = (4/3) * U / T.
    
    # Substituting the expression for U:
    # S = (4/3T) * (8 * pi^5 * k_B^4 / (15 * h^3 * c^3)) * V * T^4
    # S = (32 * pi^5 * k_B^4 / (45 * h^3 * c^3)) * V * T^3
    
    S = (sp.Rational(4, 3)) * U / T

    print("--- Equilibrium Entropy (S) ---")
    print("The entropy S is derived from the thermodynamic relation S = (4/3) * U / T.")
    print("This relationship is specific to a gas of massless relativistic particles like photons.\n")
    print("The final equation for Entropy (S) is:")
    # Use sp.simplify to ensure the expression is in its most compact form.
    sp.pprint(sp.simplify(S), use_unicode=True)


if __name__ == '__main__':
    solve_photon_gas_equilibrium()
