def print_density_profile_equation():
    """
    Prints the analytical solution for the density profile of a Mach sqrt(3) shock wave.
    """
    
    # Description of variables in the equation
    variable_description = (
        "The analytical solution for the density profile is given by the following implicit equation:\n\n"
        "Variables:\n"
        "  ρ̃ = ρ/ρ₀ : Density normalized by the ambient (pre-shock) density.\n"
        "  X = x/L   : Position normalized by the characteristic length scale L = κ/(ρ₀ M c₀ Cᵥ).\n\n"
        "The equation relates the normalized density (ρ̃) to the normalized position (X):\n"
    )
    
    # The equation is derived from the integration of the governing ODE for the specific
    # conditions (monatomic gas, Pr=3/4, M=sqrt(3)).
    # (20/9) * X = ln( (3 * (ρ̃ - 1)**2) / (ρ̃ * (2 - ρ̃)) )
    
    # Printing the final equation with each number explicitly shown
    equation_string = "  (20 / 9) * X = ln( (3 * (ρ̃ - 1)**2) / (ρ̃ * (2 - ρ̃)) )"
    
    # Explanation of the terms
    explanation = (
        "\n\nExplanation of the numerical constants:\n"
        " - The post-shock density is 2 times the ambient density (ρ̃₁ = 2).\n"
        " - The shock is centered at X = 0, where the density is ρ̃ = 1.5.\n"
        " - The constants 20, 9, 3, 1, and 2 arise from the specific Mach number (sqrt(3)),\n"
        "   the gas properties (γ=5/3), and the integration of the governing equation.\n"
    )
    
    print(variable_description)
    print(equation_string)
    print(explanation)

# Execute the function to print the solution
print_density_profile_equation()
