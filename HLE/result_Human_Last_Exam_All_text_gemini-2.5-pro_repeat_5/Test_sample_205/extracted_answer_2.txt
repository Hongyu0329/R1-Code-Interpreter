// Program to calculate 100! on the Wuxing computer.
// This program is optimized for minimal variable memory usage.

void main() {
    // Variable Declarations
    digit result[158];                  // Stores the 158 digits of 100!
    char i, j, result_size, product;    // Control and calculation variables (3D each)
    short carry;                        // Carry for multiplication (2D)

    // Initialize result to 1
    result[0] = 1;
    result_size = 1;

    // Multiply by numbers from 2 to 100
    for (i = 2; i <= 100; i++) {
        carry = 0;
        
        // Multiply the current result number by i
        for (j = 0; j < result_size; j++) {
            product = result[j] * i + carry;
            result[j] = product % 10;
            carry = product / 10;
        }

        // Handle any remaining carry by extending the result array
        while (carry > 0) {
            result[result_size] = carry % 10;
            carry = carry / 10;
            result_size++;
        }
    }

    // Print the final result (digits are stored in reverse order)
    printf("100! = ");
    j = result_size;
    while (j > 0) {
        j--;
        printf("%d", result[j]);
    }
    printf("\n");
}
