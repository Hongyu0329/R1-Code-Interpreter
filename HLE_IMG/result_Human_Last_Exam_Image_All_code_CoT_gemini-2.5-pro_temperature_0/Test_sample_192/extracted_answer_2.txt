def generate_xrd_index_table():
    """
    This function prints a table of indexed XRD peaks for CuO nanoparticles.
    The peak positions are estimated from the provided graph, and the (h k l) indices
    are based on the standard diffraction pattern for monoclinic CuO (Tenorite).
    """

    # Data for prominent peaks: (Approximate 2-theta, Miller Indices (h k l))
    # Based on standard JCPDS card no. 48-1548 for monoclinic CuO.
    peak_data = [
        {"2theta": "~32.5", "hkl": "(110)"},
        {"2theta": "~35.5", "hkl": "(-111)"},
        {"2theta": "~38.7", "hkl": "(111)"},
        {"2theta": "~48.7", "hkl": "(-202)"},
        {"2theta": "~53.4", "hkl": "(020)"},
        {"2theta": "~58.3", "hkl": "(202)"}
    ]

    # Print the table header
    print("Indexing of Prominent XRD Peaks for CuO Nanoparticles")
    print("-" * 50)
    print(f"| {'2Î¸ (degree)':<20} | {'Miller Index (h k l)':<25} |")
    print("-" * 50)

    # Print each row of the table
    for peak in peak_data:
        # Each number and character in the final table is explicitly printed here.
        theta_val = peak["2theta"]
        hkl_val = peak["hkl"]
        print(f"| {theta_val:<20} | {hkl_val:<25} |")

    print("-" * 50)

if __name__ == "__main__":
    generate_xrd_index_table()
